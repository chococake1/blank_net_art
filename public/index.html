const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); // Set canvas width and height canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Function to sort pixels around the middle of the canvas function sortPixels() { const squareSize = 100; const centerX = canvas.width / 2; const centerY = canvas.height / 2; // Sort pixels within the square centered at the middle of the canvas const imageData = ctx.getImageData(centerX - squareSize/2, centerY - squareSize/2, squareSize, squareSize); const data = imageData.data; const sortedData = Array.from(data).sort((a, b) => a - b); ctx.putImageData(new ImageData(new Uint8ClampedArray(sortedData), squareSize, squareSize), centerX - squareSize/2, centerY - squareSize/2); } // Function to animate and sort pixels every frame function animate() { // Clear canvas ctx.clearRect(0, 0, canvas.width, canvas.height); // Draw something here if needed // Sort pixels around the middle of the canvas sortPixels(); requestAnimationFrame(animate); } animate();
